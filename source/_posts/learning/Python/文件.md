```python
def size_format(size):
    if size < 1e3:
        return '%.1fB' % size
    elif size < 1e6:
        return '%.1fKB' % (size / 1e3)
    elif size < 1e9:
        return '%.1fMB' % (size / 1e6)
    elif size < 1e12:
        return '%.1fGB' % (size / 1e9)
    elif size < 1e15:
        return '%.1fTB' % (size / 1e12)
```
